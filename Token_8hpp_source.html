126enum class TokenFlags : uint8_t {
127 None = 0,
128 AtStartOfLine = 1 << 0, // Token is at the start of a line
129 HasLeadingSpace = 1 << 1, // Token has leading whitespace
130 NeedsCleaning = 1 << 2, // Token text needs cleaning (escapes, etc.)
131 IsKeyword = 1 << 3, // Token is a keyword
132};

134inline TokenFlags operator|(TokenFlags lhs, TokenFlags rhs) {
135 return static_cast<TokenFlags>(static_cast<uint8_t>(lhs) |
136 static_cast<uint8_t>(rhs));
137}

139inline TokenFlags operator&(TokenFlags lhs, TokenFlags rhs) {

141 static_cast<uint8_t>(rhs));
142}

149class Token {
150public:
151 Token() : kind(TokenKind::Unknown), flags(TokenFlags::None) {}

153 Token(TokenKind kind, SourceLocation loc, uint32_t length)

156 Token(TokenKind kind, SourceLocation loc, uint32_t length,
157 InternedString text)
158 : kind(kind), flags(TokenFlags::None), location(loc), length(length),
159 value(text) {}
160
161 // Getters
162 TokenKind getKind() const { return kind; }
163 SourceLocation getLocation() const { return location; }
164 uint32_t getLength() const { return length; }
165 TokenFlags getFlags() const { return flags; }
166 InternedString getText() const { return value; }
167
168 // Setters
169 void setKind(TokenKind kind) { this->kind = kind; }
170 void setLocation(SourceLocation loc) { location = loc; }
171 void setLength(uint32_t length) { this->length = length; }
172 void setFlags(TokenFlags flags) { this->flags = flags; }
173 void setText(InternedString text) { value = text; }
174
175 // Flag operations
176 bool hasFlag(TokenFlags flag) const {
177 return (flags & flag) != TokenFlags::None;
178 }
179
180 void addFlag(TokenFlags flag) { flags |= flag; }
181
182 void removeFlag(TokenFlags flag) {
183 flags = static_cast<TokenFlags>(static_cast<uint8_t>(flags) &
184 ~static_cast<uint8_t>(flag));
185 }
186
187 // Convenience methods
188 bool isAtStartOfLine() const { return hasFlag(TokenFlags::AtStartOfLine); }
189 bool hasLeadingSpace() const { return hasFlag(TokenFlags::HasLeadingSpace); }
190 bool isKeyword() const { return hasFlag(TokenFlags::IsKeyword); }
191 bool isLiteral() const {
192 return kind >= TokenKind::Integer && kind <= TokenKind::Boolean;
193 }
194 bool isOperator() const {
195 return kind >= TokenKind::Plus && kind <= TokenKind::MinusMinus;
196 }
197 bool isPunctuation() const {
198 return kind >= TokenKind::LeftParen && kind <= TokenKind::Hash;
199 }
200 bool isIdentifierOrKeyword() const {
201 return kind == TokenKind::Identifier ||
202 (kind >= TokenKind::Auto && kind <= TokenKind::While);
203 }
204
205 // Get source range
206 SourceRange getSourceRange() const {
207 if (length == 0)
208 return SourceRange(location);
209 // This assumes we can advance source locations
210 return SourceRange(location, SourceLocation::getFromRawEncoding(
211 location.getRawEncoding() + length));
212 }
213
214 // Comparison
215 bool operator==(const Token &other) const {
216 return kind == other.kind && location == other.location &&
217 length == other.length;
218 }
219
220 bool operator!=(const Token &other) const { return !(*this == other); }
221
222 // Check if token matches a specific kind
223 bool is(TokenKind kind) const { return this->kind == kind; }
224
225 template <typename... Kinds>
226 bool isOneOf(TokenKind kind1, Kinds... kinds) const {
227 return is(kind1) || isOneOf(kinds...);
228 }
229
230 bool isOneOf() const { return false; }
231
232 // Check if token is not a specific kind
233 bool isNot(TokenKind kind) const { return this->kind != kind; }
234
235 template <typename... Kinds> bool isNotOneOf(Kinds... kinds) const {
236 return !isOneOf(kinds...);
237 }
238
239private:
240 TokenKind kind;
241 TokenFlags flags;
242 SourceLocation location;
243 uint32_t length;
244 InternedString value; // For identifiers, literals, etc.
245};
