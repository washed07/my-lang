42void DiagnosticManager::report(const Diagnostic &diag) {
43 const DiagnosticInfo &info = getDiagnosticInfo(diag.getID());
44
45 // Check if we should suppress this diagnostic
46 if (shouldSuppress(info)) {
47 return;
48 }
49
50 // Treat warnings as errors if configured
51 DiagnosticLevel effectiveLevel = info.Level;
52 if (warningsAsErrors && info.Level == DiagnosticLevel::Warning) {
53 effectiveLevel = DiagnosticLevel::Error;
54 }
55
56 // Update statistics
57 updateStats(effectiveLevel);
58
59 // Check if we've hit the error limit
60 if (maxErrors > 0 && stats.errorCount >= maxErrors) {
61 return;
62 }
63
64 // Create a modified info with the effective level
65 DiagnosticInfo effectiveInfo = info;
66 effectiveInfo.Level = effectiveLevel;
67
68 // Report to all consumers
69 for (auto &consumer : consumers) {
70 consumer->handleDiagnostic(diag, effectiveInfo, srcMgr);
71 }
72}

193void TextDiagnosticConsumer::handleDiagnostic(const Diagnostic &diag,
194 const DiagnosticInfo &info,
195 const SourceManager *srcMgr) {
196 // Format the location
197 std::string location = "<unknown>";
198 if (srcMgr && diag.getLocation().isValid()) {
199 auto fullLoc = srcMgr->getFullLoc(diag.getLocation());
200 if (fullLoc.isValid()) {
201 std::ostringstream oss;
202 oss << fullLoc.getFilename() << ":" << fullLoc.getLineNumber() << ":"
203 << fullLoc.getColumnNumber();
204 location = oss.str();
205 }
206 }
207
208 // Format the level
209 std::string levelStr;
210 std::string colorStart, colorEnd;
211
212 if (showColor) {
213 colorEnd = "\033[0m"; // Reset color
214 }
215
216 switch (info.Level) {
217 case DiagnosticLevel::Note:
218 levelStr = "note";
219 if (showColor)
220 colorStart = "\033[36m"; // Cyan
221 break;
222 case DiagnosticLevel::Warning:
223 levelStr = "warning";
224 if (showColor)
225 colorStart = "\033[33m"; // Yellow
226 break;
227 case DiagnosticLevel::Error:
228 levelStr = "error";
229 if (showColor)
230 colorStart = "\033[31m"; // Red
231 break;
232 case DiagnosticLevel::Fatal:
233 levelStr = "fatal error";
234 if (showColor)
235 colorStart = "\033[1;31m"; // Bold red
236 break;
237 }
238
239 // Format the message
240 std::string message =
241 formatMessage(info.DetailedMessage, diag.getArguments());
242
243 // Output the diagnostic
244 os << location << ": " << colorStart << levelStr << colorEnd << ": "
245 << message << "\n";
246
247 // Print source line and highlighting if available
248 if (srcMgr && diag.getLocation().isValid()) {
249 printSourceLine(srcMgr, diag.getLocation(), diag.getRanges());
250 }
251
252 // Print fix-it hints
253 for (const auto &hint : diag.getFixItHints()) {
254 os << " fix-it: replace with '" << hint.replacement << "'\n";
255 }
256}

258void TextDiagnosticConsumer::printSourceLine(
259 const SourceManager *srcMgr, SourceLocation loc,
260 const std::vector<SourceRange> &ranges) {
261 if (!srcMgr || loc.isInvalid())
262 return;
263
264 auto fullLoc = srcMgr->getFullLoc(loc);
265 if (!fullLoc.isValid())
266 return;
267
268 // Get the line text (this is a simplified version)
269 // In a real implementation, you'd extract the actual line from the source
270 const char *lineStart = fullLoc.getData();
271 if (!lineStart)
272 return;
273
274 // Find the beginning of the line
275 while (lineStart >
276 srcMgr->getFileEntry(fullLoc.getFileID())->getBufferStart() &&
277 lineStart[-1] != '\n') {
278 --lineStart;
279 }
280
281 // Find the end of the line
282 const char *lineEnd = lineStart;
283 while (*lineEnd && *lineEnd != '\n' && *lineEnd != '\r') {
284 ++lineEnd;
285 }
286
287 // Print the line
288 std::string line(lineStart, lineEnd);
289 os << line << "\n";
290
291 // Print highlighting
292 uint32_t colNum = fullLoc.getColumnNumber();
293 if (colNum > 0) {
294 std::string highlight(colNum - 1, ' ');
295 highlight += '^';
296
297 // Add tildes for ranges
298 for (const auto &range : ranges) {
299 if (srcMgr->getFileID(range.getBegin()) == fullLoc.getFileID()) {
300 uint32_t startCol = srcMgr->getColumnNumber(range.getBegin());
301 uint32_t endCol = srcMgr->getColumnNumber(range.getEnd());
302
303 if (startCol > 0 && endCol > startCol) {
304 for (uint32_t i = startCol; i < endCol && i < highlight.size(); ++i) {
305 if (highlight[i - 1] == ' ') {
306 highlight[i - 1] = '~';
307 }
308 }
309 }
310 }
311 }
312
313 if (showColor) {
314 os << "\033[32m" << highlight << "\033[0m\n"; // Green
315 } else {
316 os << highlight << "\n";
317 }
318 }
319}

321std::string
322TextDiagnosticConsumer::formatMessage(std::string_view message,
323 const std::vector<std::string> &args) {
324 std::string result(message);
325
326 // Replace %0, %1, etc. with arguments
327 for (size_t i = 0; i < args.size(); ++i) {
328 std::string placeholder = "%" + std::to_string(i);
329 size_t pos = 0;
330 while ((pos = result.find(placeholder, pos)) != std::string::npos) {
331 result.replace(pos, placeholder.length(), args[i]);
332 pos += args[i].length();
333 }
334 }
335
336 return result;
337}
