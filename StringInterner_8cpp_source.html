15// SIMD-optimized string comparison
16static bool fast_string_equal(const char *a, const char *b, size_t len) {
17#if defined(__AVX2__) || (defined(_MSC_VER) && defined(__AVX2__))
18 // Use AVX2 for chunks of 32 bytes
19 while (len >= 32) {
20 __m256i chunk_a = _mm256_loadu_si256(reinterpret_cast<const __m256i *>(a));
21 __m256i chunk_b = _mm256_loadu_si256(reinterpret_cast<const __m256i *>(b));
22 __m256i cmp = _mm256_cmpeq_epi8(chunk_a, chunk_b);
23
24 if (_mm256_movemask_epi8(cmp) != static_cast<int>(0xFFFFFFFF)) {
25 return false;
26 }
27
28 a += 32;
29 b += 32;
30 len -= 32;
31 }
32#elif defined(__SSE2__) || (defined(_MSC_VER) && defined(__SSE2__))
33 // Use SSE2 for chunks of 16 bytes
34 while (len >= 16) {
35 __m128i chunk_a = _mm_loadu_si128(reinterpret_cast<const __m128i *>(a));
36 __m128i chunk_b = _mm_loadu_si128(reinterpret_cast<const __m128i *>(b));
37 __m128i cmp = _mm_cmpeq_epi8(chunk_a, chunk_b);
38
39 if (_mm_movemask_epi8(cmp) != 0xFFFF) {
40 return false;
41 }
42
43 a += 16;
44 b += 16;
45 len -= 16;
46 }
47#endif
48
49 // Handle remaining bytes
50 return std::memcmp(a, b, len) == 0;
51}

56StringInterner::StringStorage::StringStorage(std::string_view str,
57 ArenaAllocator *arena)
58 : size(str.size()), usesArena(arena != nullptr) {
59 // Allocate memory either from arena or heap
60 if (arena) {
61 // Use arena allocation for better memory locality
62 data = arena->allocateString(str.data(), str.size());
63 if (!data) {
64 throw std::bad_alloc();
65 }
66 } else {
67 // Use aligned heap allocation for SIMD operations
68 size_t aligned_size = (size + 16) & ~15;
69 auto heap_data = std::make_unique<char[]>(aligned_size);
70
71 // Use fast copy for larger strings
72 if (size >= 32) {
73 std::memcpy(heap_data.get(), str.data(), size);
74 } else {
75 // Manual copy for small strings to avoid function call overhead
76 const char *src = str.data();
77 char *dst = heap_data.get();
78 for (size_t i = 0; i < size; ++i) {
79 dst[i] = src[i];
80 }
81 }
82 heap_data[size] = '\0'; // Null terminate
83
84 // Transfer ownership to raw pointer
85 data = heap_data.release();
86 }
87}

158InternedString StringInterner::intern(std::string_view str) {
159 ++stats.lookupCount;
160
161 // Early exit for empty strings
162 if (str.empty()) {
163 static const char empty_str[] = "";
164 return InternedString(empty_str);
165 }
166
167 // Early optimization: compute string length once (for potential future use)
168 // size_t str_length = str.length();
169
170 // Fast path: check if already interned (shared lock)
171 {
172 std::shared_lock<std::shared_mutex> lock(Mutex);
173 auto it = LookupMap.find(str);
174 if (it != LookupMap.end()) {
175 return InternedString(it->second);
176 }
177 }
178
179 // Slow path: need to intern the string (exclusive lock)
180 std::unique_lock<std::shared_mutex> lock(Mutex);
181
182 // Double-check that another thread didn't intern it while we were waiting
183 auto it = LookupMap.find(str);
184 if (it != LookupMap.end()) {
185 return InternedString(it->second);
186 }
187
188 // Create new storage for the string (using arena if available)
189 auto storage = std::make_unique<StringStorage>(str, arenaAllocator);
190 const char *ptr = storage->c_str();
191
192 // Reserve capacity if needed to avoid rehashing
193 if (Storage.size() >= Storage.bucket_count() * 0.75) {
194 Storage.reserve(Storage.size() * 2);
195 LookupMap.reserve(LookupMap.size() * 2);
196 }
197
198 // Add to storage set
199 auto [storageIt, storageInserted] = Storage.insert(std::move(storage));
200 if (!storageInserted) {
201 // This should never happen with our current implementation
202 ++stats.collisionCount;
203 return InternedString((*storageIt)->c_str());
204 }
205
206 // Add to lookup map using the stored string as the key
207 std::string_view storedView(ptr, str.size());
208 LookupMap[storedView] = ptr;
209
210 // Update statistics efficiently
211 ++stats.internCount;
212 ++stats.uniqueStringCount;
213 size_t string_memory = str.size() + 1; // +1 for null terminator
214 stats.memoryUsedCount += string_memory;
215
216 // Update average length using incremental formula
217 stats.averageLength =
218 (stats.averageLength * (stats.uniqueStringCount - 1) + str.size()) /
219 stats.uniqueStringCount;
220
221 return InternedString(ptr);
222}
