42void *ArenaAllocator::allocate(size_t size, size_t alignment) {
43 if (size == 0) {
44 return nullptr;
45 }
46
47 if (size > kMaxAllocationSize) {
48 // For very large allocations, we could fall back to regular heap allocation
49 // but for now, we'll just fail
50 return nullptr;
51 }
52
53 // Ensure minimum alignment
54 if (alignment < kDefaultAlignment) {
55 alignment = kDefaultAlignment;
56 }
57
58 // Try to allocate from the current (last) chunk
59 if (!Chunks.empty()) {
60 size_t oldUsed = Chunks.back().used;
61 void *ptr = Chunks.back().allocate(size, alignment);
62 if (ptr) {
63 size_t actualAllocated = Chunks.back().used - oldUsed;
64 updateStats(size, actualAllocated);
65 return ptr;
66 }
67 }
68
69 // Need a new chunk
70 size_t neededSize = size + alignment - 1; // Worst case alignment padding
71 allocateNewChunk(std::max(neededSize, chunkSize));
72
73 // Try again with the new chunk
74 assert(!Chunks.empty());
75 size_t oldUsed = Chunks.back().used;
76 void *ptr = Chunks.back().allocate(size, alignment);
77
78 if (ptr) {
79 size_t actualAllocated = Chunks.back().used - oldUsed;
80 updateStats(size, actualAllocated);
81 return ptr;
82 }
83
84 // This should never happen if our math is correct
85 return nullptr;
86}
