190std::pair<std::shared_ptr<FileEntry>, std::error_code>
191FileManager::loadFile(const std::string &filename) {
192 // Get file info first
193 std::error_code ec;
194 if (!std::filesystem::exists(filename, ec) || ec) {
195 return {nullptr,
196 ec ? ec
197 : std::make_error_code(std::errc::no_such_file_or_directory)};
198 }
199
200 auto fileSize = std::filesystem::file_size(filename, ec);
201 if (ec) {
202 return {nullptr, ec};
203 }
204
205 auto modTime = std::filesystem::last_write_time(filename, ec);
206 if (ec) {
207 return {nullptr, ec};
208 }
209
210 time_t modTimeT = std::chrono::duration_cast<std::chrono::seconds>(
211 modTime.time_since_epoch())
212 .count();
213
214 ++stats.fileOpenCount;
215 stats.bytesReadCount += fileSize;
216
217 // Try memory mapping for large files
218 if (memoryMappingEnabled && fileSize >= memoryMappingThreshold) {
219 // For simplicity, fall back to regular file reading
220 // A full implementation would use platform-specific memory mapping
221 }
222
223 // Regular file reading
224 std::ifstream file(filename, std::ios::binary);
225 if (!file) {
226 return {nullptr, std::make_error_code(std::errc::io_error)};
227 }
228
229 // Allocate buffer with extra space for null terminator
230 auto buffer = std::make_unique<char[]>(fileSize + 1);
231
232 file.read(buffer.get(), fileSize);
233 if (file.gcount() != static_cast<std::streamsize>(fileSize)) {
234 return {nullptr, std::make_error_code(std::errc::io_error)};
235 }
236
237 // Null-terminate the buffer
238 buffer[fileSize] = '\0';
239
240 InternedString internedFilename = interner.intern(filename);
241 auto entry = std::make_shared<FileEntry>(internedFilename, std::move(buffer),
242 fileSize, modTimeT);
243
244 return {entry, {}};
245}

247void FileManager::evictIfNeeded() {
248 // Simple LRU eviction based on cache size
249 // In a production implementation, this would be more sophisticated
250 if (maxCacheSize == SIZE_MAX) {
251 return; // No limit
252 }
253
254 size_t currentSize = 0;
255 for (const auto &[filename, entry] : fileCache) {
256 currentSize += entry->getSize();
257 }
258
259 if (currentSize <= maxCacheSize) {
260 return; // Under limit
261 }
262
263 // Simple eviction: remove files until we're under the limit
264 // A better implementation would track access times for true LRU
265 auto it = fileCache.begin();
266 while (it != fileCache.end() && currentSize > maxCacheSize) {
267 currentSize -= it->second->getSize();
268 it = fileCache.erase(it);
269 --stats.fileCacheCount;
270 }
271}
