237// Optimized keyword lookup using constexpr sorted array for O(log n) binary
238// search
239struct KeywordEntry {
240 std::string_view keyword;
241 TokenKind kind;
242
243 constexpr bool operator<(const KeywordEntry &other) const {
244 return keyword < other.keyword;
245 }
246 constexpr bool operator<(std::string_view str) const { return keyword < str; }
247};

277// Optimized operator precedence using direct array lookup
278static std::array<int, static_cast<size_t>(TokenKind::Count)>
279createOperatorPrecedenceTable() {
280 std::array<int, static_cast<size_t>(TokenKind::Count)> table{};
281
282 // Initialize all to 0 (not an operator)
283 std::fill(table.begin(), table.end(), 0);
284
285 // Multiplicative
286 table[static_cast<size_t>(TokenKind::Star)] = 14;
287 table[static_cast<size_t>(TokenKind::Slash)] = 14;
288 table[static_cast<size_t>(TokenKind::Percent)] = 14;
289
290 // Additive
291 table[static_cast<size_t>(TokenKind::Plus)] = 13;
292 table[static_cast<size_t>(TokenKind::Minus)] = 13;
293
294 // Shift
295 table[static_cast<size_t>(TokenKind::LesserLesser)] = 12;
296 table[static_cast<size_t>(TokenKind::GreaterGreater)] = 12;
297
298 // Relational
299 table[static_cast<size_t>(TokenKind::Less)] = 11;
300 table[static_cast<size_t>(TokenKind::LessEqual)] = 11;
301 table[static_cast<size_t>(TokenKind::Greater)] = 11;
302 table[static_cast<size_t>(TokenKind::GreaterEqual)] = 11;
303
304 // Equality
305 table[static_cast<size_t>(TokenKind::EqualEqual)] = 10;
306 table[static_cast<size_t>(TokenKind::NotEqual)] = 10;
307
308 // Bitwise AND
309 table[static_cast<size_t>(TokenKind::Amp)] = 9;
310
311 // Bitwise XOR
312 table[static_cast<size_t>(TokenKind::Caret)] = 8;
313
314 // Bitwise OR
315 table[static_cast<size_t>(TokenKind::Pipe)] = 7;
316
317 // Logical AND
318 table[static_cast<size_t>(TokenKind::AmpAmp)] = 6;
319
320 // Logical OR
321 table[static_cast<size_t>(TokenKind::PipePipe)] = 5;
322
323 // Assignment
324 table[static_cast<size_t>(TokenKind::Equal)] = 2;
325 table[static_cast<size_t>(TokenKind::PlusEqual)] = 2;
326 table[static_cast<size_t>(TokenKind::MinusEqual)] = 2;
327 table[static_cast<size_t>(TokenKind::StarEqual)] = 2;
328 table[static_cast<size_t>(TokenKind::SlashEqual)] = 2;
329 table[static_cast<size_t>(TokenKind::PercentEqual)] = 2;

355TokenKind TokenInfo::getKeywordKind(std::string_view identifier) {
356 // Fast binary search on sorted array
357 auto it = std::lower_bound(
358 KEYWORD_LOOKUP_TABLE.begin(), KEYWORD_LOOKUP_TABLE.end(), identifier,
359 [](const KeywordEntry &entry, std::string_view str) {
360 return entry.keyword < str;
361 });
362
363 if (it != KEYWORD_LOOKUP_TABLE.end() && it->keyword == identifier) {
364 return it->kind;
365 }
366 return TokenKind::Identifier;
367}

381int TokenInfo::getOperatorPrecedence(TokenKind kind) {
382 static const auto OPERATOR_PRECEDENCE = createOperatorPrecedenceTable();
383 size_t index = static_cast<size_t>(kind);
384 if (index >= OPERATOR_PRECEDENCE.size()) {
385 return 0;
386 }
387 return OPERATOR_PRECEDENCE[index];
388}

390bool TokenInfo::isLeftAssociative(TokenKind kind) {
391 // Most operators are left associative
392 return getOperatorPrecedence(kind) > 0 && kind != TokenKind::Equal &&
393 kind != TokenKind::PlusEqual && kind != TokenKind::MinusEqual &&
394 kind != TokenKind::StarEqual && kind != TokenKind::SlashEqual &&
395 kind != TokenKind::PercentEqual;
396}
397
398bool TokenInfo::isRightAssociative(TokenKind kind) {
399 // Assignment operators are right associative
400 return kind == TokenKind::Equal || kind == TokenKind::PlusEqual ||
401 kind == TokenKind::MinusEqual || kind == TokenKind::StarEqual ||
402 kind == TokenKind::SlashEqual || kind == TokenKind::PercentEqual;
403}
