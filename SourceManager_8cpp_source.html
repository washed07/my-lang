132 std::lock_guard<std::mutex> lock(stateMutex);

134 InternedString filename = entry->getFilename();

197FileID SourceManager::getFileID(SourceLocation loc) const {
198 if (loc.isInvalid()) {
199 return FileID::getInvalidID();
200 }
201
202 uint32_t locID = loc.getRawEncoding();
203
204 // Fast path: check cache for recently accessed location
205 if (g_location_cache.last_location.isValid() &&
206 g_location_cache.last_file_id.isValid()) {
207
208 const FileInfo &cached_info =
209 loadedFiles[g_location_cache.last_file_id.getHashValue() - 1];
210 uint32_t file_start = cached_info.offset;
211 uint32_t file_end =
212 file_start + static_cast<uint32_t>(cached_info.entry->getSize());
213
214 if (locID >= file_start && locID <= file_end) {
215 return g_location_cache.last_file_id;
216 }
217 }
218
219 // Binary search to find the file containing this location
220 auto it = std::upper_bound(
221 loadedFiles.begin(), loadedFiles.end(), locID,
222 [](uint32_t loc, const FileInfo &info) { return loc < info.offset; });
223
224 if (it == loadedFiles.begin()) {
225 return FileID::getInvalidID();
226 }
227
228 --it;
229
230 // Check if the location is within this file's range
231 uint32_t fileStart = it->offset;
232 uint32_t fileEnd = fileStart + static_cast<uint32_t>(it->entry->getSize());
233
234 if (locID < fileStart || locID > fileEnd) {
235 return FileID::getInvalidID();
236 }
237
238 // Convert iterator to FileID and cache result
239 size_t index = std::distance(loadedFiles.begin(), it);
240 FileID result = FileID(static_cast<uint32_t>(index + 1));
241
242 // Update cache
243 g_location_cache.last_location = loc;
244 g_location_cache.last_file_id = result;
245
246 return result;
247}

293uint32_t SourceManager::getLineNumber(SourceLocation loc) const {
294 if (loc.isInvalid()) {
295 return 0;
296 }
297
298 // Fast path: check if we can use cached line info
299 if (g_location_cache.last_location.isValid() &&
300 g_location_cache.last_file_id.isValid() &&
301 g_location_cache.last_line > 0) {
302
303 // If it's the same location, return cached result
304 if (loc == g_location_cache.last_location) {
305 return g_location_cache.last_line;
306 }
307
308 // If it's in the same file and on the same line, we can compute quickly
309 FileID fid = getFileID(loc);
310 if (fid == g_location_cache.last_file_id &&
311 g_location_cache.last_line_start) {
312 uint32_t offset = getFileOffset(loc);
313 const FileInfo &info = loadedFiles[fid.getHashValue() - 1];
314
315 // Check if still on the same line
316 const char *current_pos = info.entry->getBufferStart() + offset;
317 const char *line_start = g_location_cache.last_line_start;
318
319 // Scan forward to see if we hit a newline
320 const char *scan = line_start;
321 while (scan <= current_pos && *scan != '\n' && *scan != '\r') {
322 ++scan;
323 }
324
325 if (scan > current_pos) {
326 // Still on same line
327 return g_location_cache.last_line;
328 }
329 }
330 }
331
332 FileID fid = getFileID(loc);
333 if (fid.isInvalid()) {
334 return 0;
335 }
336
337 computeLineOffsets(fid);
338
339 const FileInfo &info = loadedFiles[fid.getHashValue() - 1];
340 uint32_t offset = getFileOffset(loc);
341
342 uint32_t line = findLineNumber(info.lineOffsets, offset);
343
344 // Update cache
345 g_location_cache.last_location = loc;
346 g_location_cache.last_file_id = fid;
347 g_location_cache.last_line = line;
348
349 // Cache line start position for next lookup
350 if (!info.lineOffsets.empty() && line > 0 &&
351 line <= info.lineOffsets.size()) {
352 uint32_t line_offset = (line == 1) ? 0 : info.lineOffsets[line - 2] + 1;
353 g_location_cache.last_line_start =
354 info.entry->getBufferStart() + line_offset;
355 }
356
357 return line;
358}

361 if (loc.isInvalid()) {
362 return 0;
363 }
364
365 // Fast path: check cache
366 if (g_location_cache.last_location.isValid() &&
367 g_location_cache.last_file_id.isValid() &&
368 g_location_cache.last_column > 0) {
369
370 if (loc == g_location_cache.last_location) {
371 return g_location_cache.last_column;

374
375 FileID fid = getFileID(loc);
376 if (fid.isInvalid()) {

395 g_location_cache.last_file_id = fid;

544void SourceManager::computeLineOffsets(FileID fid) const {
545 if (fid.isInvalid() || fid.getHashValue() == 0 ||
546 fid.getHashValue() > loadedFiles.size()) {
547 return;
548 }
549
550 FileInfo &info = const_cast<FileInfo &>(loadedFiles[fid.getHashValue() - 1]);
551
552 if (info.lineOffsetsComputed) {
553 return;
554 }
555
556 ++stats.lineComputationCount;
557
558 // Compute line start offsets
559 const char *data = info.entry->getBufferStart();
560 size_t size = info.entry->getSize();
561
562 info.lineOffsets.clear();
563 info.lineOffsets.reserve(size / 40 + 16); // Estimate ~40 chars per line
564 info.lineOffsets.push_back(0); // Line 1 starts at offset 0
565
566 // SIMD-optimized newline scanning
567#if defined(__AVX2__) || (defined(_MSC_VER) && defined(__AVX2__))
568 const __m256i newline = _mm256_set1_epi8('\n');
569 size_t i = 0;
570
571 // Process 32 bytes at a time with AVX2
572 while (i + 32 <= size) {
573 __m256i chunk =
574 _mm256_loadu_si256(reinterpret_cast<const __m256i *>(data + i));
575 __m256i cmp = _mm256_cmpeq_epi8(chunk, newline);
576 uint32_t mask = _mm256_movemask_epi8(cmp);
577
578 if (mask != 0) {
579 // Found newlines, process each bit
580 for (int bit = 0; bit < 32; ++bit) {
581 if (mask & (1u << bit)) {
582 info.lineOffsets.push_back(static_cast<uint32_t>(i + bit + 1));
583 }
584 }
585 }
586 i += 32;
587 }
588
589 // Handle remaining bytes
590 while (i < size) {
591 if (data[i] == '\n') {
592 info.lineOffsets.push_back(static_cast<uint32_t>(i + 1));
593 }
594 ++i;
595 }
596
597#elif defined(__SSE2__) || (defined(_MSC_VER) && defined(__SSE2__))
598 const __m128i newline = _mm_set1_epi8('\n');
599 size_t i = 0;
600
601 // Process 16 bytes at a time with SSE2
602 while (i + 16 <= size) {
603 __m128i chunk =
604 _mm_loadu_si128(reinterpret_cast<const __m128i *>(data + i));
605 __m128i cmp = _mm_cmpeq_epi8(chunk, newline);
606 uint32_t mask = _mm_movemask_epi8(cmp);
607
608 if (mask != 0) {
609 // Found newlines, process each bit
610 for (int bit = 0; bit < 16; ++bit) {
611 if (mask & (1u << bit)) {
612 info.lineOffsets.push_back(static_cast<uint32_t>(i + bit + 1));
613 }
614 }
615 }
616 i += 16;
617 }
618
619 // Handle remaining bytes
620 while (i < size) {
621 if (data[i] == '\n') {
622 info.lineOffsets.push_back(static_cast<uint32_t>(i + 1));
623 }
624 ++i;
625 }
626
627#else
628 // Fallback scalar implementation with unrolled loop
629 size_t i = 0;
630 while (i + 8 <= size) {
631 // Unroll loop for better performance
632 if (data[i] == '\n')
633 info.lineOffsets.push_back(static_cast<uint32_t>(i + 1));
634 if (data[i + 1] == '\n')
635 info.lineOffsets.push_back(static_cast<uint32_t>(i + 2));
636 if (data[i + 2] == '\n')
637 info.lineOffsets.push_back(static_cast<uint32_t>(i + 3));
638 if (data[i + 3] == '\n')
639 info.lineOffsets.push_back(static_cast<uint32_t>(i + 4));
640 if (data[i + 4] == '\n')
641 info.lineOffsets.push_back(static_cast<uint32_t>(i + 5));
642 if (data[i + 5] == '\n')
643 info.lineOffsets.push_back(static_cast<uint32_t>(i + 6));
644 if (data[i + 6] == '\n')
645 info.lineOffsets.push_back(static_cast<uint32_t>(i + 7));
646 if (data[i + 7] == '\n')
647 info.lineOffsets.push_back(static_cast<uint32_t>(i + 8));
648 i += 8;
649 }
650
651 // Handle remaining bytes
652 while (i < size) {
653 if (data[i] == '\n') {
654 info.lineOffsets.push_back(static_cast<uint32_t>(i + 1));
655 }
656 ++i;
657 }
658#endif
659
660 info.lineOffsetsComputed = true;
661}

663uint32_t SourceManager::findLineNumber(const std::vector<uint32_t> &lineOffsets,
664 uint32_t offset) const {
